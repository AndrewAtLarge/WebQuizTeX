\documentclass{mathquiz}
\title{A multiple choice example with zero or more correct answers}
\begin{document}
  \begin{question}     % a quiz question
     Which of the following integers are prime?
     % multiple (0 or more) correct answers rendered in two columns
     \begin{choice}[multiple]
       \correct 17   \response No, $1^3=1$
       \incorrect 12 \response No, $12 = 3\times 4$
       \incorrect 27 \response No, $27=3^3$
       \correct 91   \response Yes, $4^3=64$
       \incorrect 0  \response Prime numbers are non-zero
       \incorrect 1  \response One is not a prime number because it is invertible
       \incorrect 7  \response No, $7^3=343$
       \incorrect 8  \response No, $8^3=512$
       \incorrect 9  \response No, $9^3=729$
     \end{choice}
  \end{question}
\end{document}
